function msgpackDecode(buf) {
    let offset = 0;

    function read() {
        const byte = buf[offset++];

        // positive fixint (0x00 - 0x7f)
        if (byte <= 0x7f) return byte;

        // fixstr (0xa0 - 0xbf)
        if (byte >= 0xa0 && byte <= 0xbf) {
            const len = byte & 0x1f;
            return String.fromCharCode(...buf.subarray(offset, offset += len));
        }

        // fixmap (0x80 - 0x8f)
        if (byte >= 0x80 && byte <= 0x8f) {
            const len = byte & 0x0f;
            const obj = {};
            for (let i = 0; i < len; i++) {
                const key = read();
                obj[key] = read();
            }
            return obj;
        }

        // fixarray (0x90 - 0x9f)
        if (byte >= 0x90 && byte <= 0x9f) {
            const len = byte & 0x0f;
            return Array.from({ length: len }, read);
        }

        switch (byte) {
            case 0xc0: return null;
            case 0xc2: return false;
            case 0xc3: return true;

            // uint 8/16/32
            case 0xcc: return buf[offset++];
            case 0xcd: return (buf[offset++] << 8) | buf[offset++];
            case 0xce: return ((buf[offset++] << 24) | (buf[offset++] << 16) | (buf[offset++] << 8) | buf[offset++]) >>> 0;

            // int 8/16/32
            case 0xd0: return buf[offset++] - (buf[offset-1] & 0x80 ? 256 : 0);
            case 0xd1: { const v = (buf[offset++] << 8) | buf[offset++]; return v & 0x8000 ? v - 65536 : v; }
            case 0xd2: { const v = (buf[offset++] << 24) | (buf[offset++] << 16) | (buf[offset++] << 8) | buf[offset++]; return v; }

            // str 8/16
            case 0xd9: { const len = buf[offset++]; return String.fromCharCode(...buf.subarray(offset, offset += len)); }
            case 0xda: { const len = (buf[offset++] << 8) | buf[offset++]; return String.fromCharCode(...buf.subarray(offset, offset += len)); }

            // negative fixint (0xe0 - 0xff)
            default:
                if (byte >= 0xe0) return byte - 256;
                throw new Error(`Unsupported msgpack byte: 0x${byte.toString(16)}`);
        }
    }

    return read();
}
